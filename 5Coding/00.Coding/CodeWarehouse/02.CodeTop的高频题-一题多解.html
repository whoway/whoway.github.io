void BFS( TreeNode * root)
{
    if( nullptr==root )
    {
        help.push_back( -1 );
        return ;
    }
    queue< TreeNode * > Q;
    Q.push( root );

    while( !Q.empty() )
    {
        int Len=Q.size();
        for(int i=0; i<Len; ++i)
        {
            TreeNode * cur=Q.front();
            if( nullptr==cur )
            {
                help.push_back( -1 );
                Q.pop();
                continue;
            }

            //nullptr!=cur
            help.push_back( cur->val );
            Q.pop();
            Q.push( cur->left );
            Q.push( cur->right );
        }
    }

}
