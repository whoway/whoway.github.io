class Solution {
public:
    string reverseWords(string s) {
        //å“¨å…µ
        s.push_back(' ');
        string res;

        int pos=0;
        while( pos<s.size() )
        {
            if( ' '==s[pos] )
            {
                res+=' ';
                ++pos;
            }
            else
            {
                int endpos=s.find(' ', pos);
                string temp=s.substr( pos, endpos-pos );
                reverse( temp.begin(), temp.end() );
                res+=temp;
                pos=endpos;
            }
        }

        res.pop_back();
        return res;
    }
};
