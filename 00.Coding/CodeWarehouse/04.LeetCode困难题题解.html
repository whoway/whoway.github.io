public:
    int candy(vector<int>& ratings) {
        int n = ratings.size();
        vector<int> left(n);

        //å·¦è§„åˆ™

    }

public:
    int calculate(string s) {
        stack<int> ops;
        ops.push(1);
        int sign = 1;

        int ret = 0;
        int n = s.length();
        int i = 0;
        while (i < n) {
            if (s[i] == ' ') {

                sign = ops.top();
                i++;
            } else if (s[i] == '-') {
                sign = -ops.top();

            } else if (s[i] == '(') {
                ops.push(sign);

                    i++;
                }
                ret += sign * num;
            }
        }
        return ret;
    }

    }

                                class Solution {
public:
    int newInteger(int n) {
        stack<int> help;
        while( n )
        {
            help.push( n%9 );
            n/=9;
        }
        
        int res=0;
        while( !help.empty() )
        {
            res=res*10+help.top();
            help.pop();
        }

        return res;
    }
};
