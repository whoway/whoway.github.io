 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if( nullptr==root )
        {
            return root;
        }

        if( root==p )
        {
            return root;
        }

        if( root==q )
        {
            return root;
        }

        TreeNode * first=lowestCommonAncestor( root->left, p, q);
        TreeNode * second=lowestCommonAncestor( root->right, p , q);

        //å¦‚æœä¸¤è¾¹éƒ½æ‰¾åˆ°äº†
        if( nullptr!=first && nullptr!=second )
        {
            return root;
        }

        if( nullptr==first )
        {
            return second;
        }
        else
        {
            return first;
        }
    }
};
